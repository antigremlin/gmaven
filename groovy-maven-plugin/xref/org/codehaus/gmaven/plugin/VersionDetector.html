24      public String detectVersion(final ClassLoader classLoader) {

26          String version = getVersion(classLoader, "groovy.lang.GroovySystem", "getVersion");
27          if (version == null) {
28              // Older versions of Groovy expose the version via InvokerHelper.getVersion()

34      private String getVersion(final ClassLoader classLoader, final String className, final String methodName) {
35          try {

37              Method method = type.getMethod(methodName);
38              Object result = method.invoke(null);
39              if (result != null) {
40                  return result.toString();
41              }
42          }
43          catch (Throwable e) {
44              log.warn("Unable determine version from: {}", className);
45          }
46          return null;
