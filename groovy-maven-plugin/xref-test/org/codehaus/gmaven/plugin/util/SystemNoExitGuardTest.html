14  package org.codehaus.gmaven.plugin.util;

16  import java.io.BufferedInputStream;
17  import java.io.ByteArrayOutputStream;
18  import java.io.InputStream;
19  import java.io.PrintStream;
20  
21  import org.sonatype.sisu.litmus.testsupport.TestSupport;
22  
23  import org.codehaus.gmaven.plugin.util.SystemNoExitGuard.Task;
24  import org.junit.Test;
25  
26  import static org.hamcrest.Matchers.is;
27  import static org.junit.Assert.assertThat;
28  
29  /**
30   * Tests for {@link SystemNoExitGuard}.
31   */
32  public class SystemNoExitGuardTest
33      extends TestSupport
34  {
35    @Test
36    public void systemStreamsAreRestored() throws Exception {
37      InputStream systemIn = System.in;
38      PrintStream systemOut = System.out;
39      PrintStream systemErr = System.err;
40  
41      new SystemNoExitGuard().run(new Task()
42      {
43        @Override
44        public void run() throws Exception {
45          // change all system streams
46          System.setIn(new BufferedInputStream(System.in));
47          System.setOut(new PrintStream(new ByteArrayOutputStream()));
48          System.setErr(new PrintStream(new ByteArrayOutputStream()));
49        }
50      });
51  
52      // ensure that all system streams are reset
53      assertThat(System.in, is(systemIn));
54      assertThat(System.out, is(systemOut));
55      assertThat(System.err, is(systemErr));
56    }
57  }
