80    /**
81     * Create a {@link GroovyClassLoader} from given {@link ClassLoader} and {@link ResourceLoader}.
82     */
83    public GroovyClassLoader createGroovyClassLoader(final ClassLoader classLoader, final ResourceLoader resourceLoader) {
84      return AccessController.doPrivileged(new PrivilegedAction<GroovyClassLoader>()
85      {
86        @Override
87        public GroovyClassLoader run() {
88          GroovyClassLoader gcl = new GroovyClassLoader(classLoader);
89          gcl.setResourceLoader(createGroovyResourceLoader(resourceLoader));
90          return gcl;
91        }
92      });
93    }

95    /**
96     * Creates a {@link GroovyResourceLoader} from a {@link ResourceLoader}.
97     */
98    public GroovyResourceLoader createGroovyResourceLoader(final ResourceLoader resourceLoader) {
99      checkNotNull(resourceLoader);
100 
101     return new GroovyResourceLoader()
102     {
103       @Override
104       public URL loadGroovySource(final String name) throws MalformedURLException {
105         return resourceLoader.loadResource(name);
106       }
107     };
108   }

110   /**
111    * Creates a {@link GroovyCodeSource} from a {@link ClassSource}.
112    */
113   public GroovyCodeSource createGroovyCodeSource(final ClassSource source) throws IOException {
114     checkNotNull(source);
115 
116     if (source.getUrl() != null) {
117       return new GroovyCodeSource(source.getUrl());
118     }
119     if (source.getFile() != null) {
120       return new GroovyCodeSource(source.getFile());
121     }
122     if (source.getInline() != null) {
123       Inline inline = source.getInline();
124       return new GroovyCodeSource(inline.getInput(), inline.getName(), inline.getCodeBase());
125     }
126 
127     throw new Error("Unable to create GroovyCodeSource from: " + source);
128   }

130   /**
131    * Creates a {@link Closure} from a {@link ClosureTarget}.
132    */
133   public Closure createClosure(final Object owner, final ClosureTarget target) {
134     checkNotNull(owner);
135     checkNotNull(target);
136 
137     return new Closure(owner)
138     {
139       @Override
140       public Object call(final Object[] args) {
141         try {
142           return target.call(args);
143         }
144         catch (Exception e) {
145           throw Throwables.propagate(e);
146         }
147       }
148 
149       @Override
150       public String toString() {
151         return Closure.class.getSimpleName() + "{owner=" + owner + ", target=" + target + "}";
152       }
153     };
154   }

156   /**
157    * Create an object for a {@link MagicContext} entry.
158    */
159   public Object createMagicContextValue(final MagicContext magic) {
160     checkNotNull(magic);
161 
162     switch (magic) {
163       // Create the AntBuilder instance, normalizing its output to match mavens
164       case ANT_BUILDER: {
165         AntBuilder ant = new AntBuilder();
166         // TODO: Do we need to root the ant-project or otherwise augment the configuration?
167 
168         Object obj = ant.getAntProject().getBuildListeners().elementAt(0);
169         if (obj instanceof BuildLogger) {
170           BuildLogger logger = (BuildLogger) obj;
171           logger.setEmacsMode(true);
172         }
173         return ant;
174       }
175     }
176 
177     throw new Error("Unsupported magic context: " + magic);
178   }

180   /**
181    * Create script binding, handling conversion of {@link ClosureTarget} and {@link MagicContext} entries.
182    */
183   public Binding createBinding(final Map<String, Object> context) {
184     Binding binding = new Binding();
185 
186     log.debug("Binding:");
187     for (Entry<String, Object> entry : context.entrySet()) {
188       String key = entry.getKey();
189       Object value = entry.getValue();
190 
191       if (value instanceof ClosureTarget) {
192         value = createClosure(this, (ClosureTarget) value);
193       }
194       else if (value instanceof MagicContext) {
195         value = createMagicContextValue((MagicContext) value);
196       }
197 
198       log.debug("  {}={}", key, value);
199       binding.setVariable(entry.getKey(), value);
200     }
201 
202     return binding;
203   }
